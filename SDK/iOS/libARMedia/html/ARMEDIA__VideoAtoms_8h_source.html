67 /* Atoms :
68 LEGEND :
69 -> specific = specific function to generate this atom
70 -> empty = use atomFromData (0, "name", NULL);
71 -> from data = use atomFromData (dataSize, "name", dataPointer);
72 
73 ftyp -> specific
74 mdat -> specific (include freeAtom if needed)
75 moov -> empty
76  |- mvhd -> specific
77  |- trak -> empty
78  | |- tkhd -> specific
79  | \- mdia -> empty
80  | |- mdhd -> specific
81  | |- hdlr -> specific (for mdia)
82  | \- minf -> empty
83  | |- vmhd -> specific
84  | |- hdlr -> specific (for minf)
85  | |- dinf -> empty
86  | | \- dref -> specific
87  | \- stbl -> empty
88  | |- stsd -> specific
89  | |- stts -> from data
90  | |- stss -> from data (i frame positions as uint32_t network endian)
91  | |- stsc -> specific
92  | |- stsz -> from data (frames sizes as uint32_t network endian)
93  | \- stco -> from data (frames offset as uint32_t network endian)
94  \- udta -> empty
95  |- meta1 -> all meta specific (metadataAtomFromTagAndValue)
96  [...]
97  \- metaN -> all meta specific (metadataAtomFromTagAndValue)
98 pvat -> specific
99 */

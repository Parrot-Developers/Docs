482 static inline uint16_t ARSAL_Endianness_bswaps (uint16_t orig)
483 {
484  return ((orig & 0xFF00) >> 8) | ((orig & 0x00FF) << 8);
485 }
486 
492 static inline uint32_t ARSAL_Endianness_bswapl (uint32_t orig)
493 {
494  return ((orig & 0xFF000000) >> 24) | ((orig & 0x00FF0000) >> 8) | ((orig & 0x0000FF00) << 8) | ((orig & 0x000000FF) << 24);
495 }
496 
502 static inline uint64_t ARSAL_Endianness_bswapll (uint64_t orig)
503 {
504  return ((orig << 56) |
505  ((orig << 40) & 0xff000000000000ULL) |
506  ((orig << 24) & 0xff0000000000ULL) |
507  ((orig << 8) & 0xff00000000ULL) |
508  ((orig >> 8) & 0xff000000ULL) |
509  ((orig >> 24) & 0xff0000ULL) |
510  ((orig >> 40) & 0xff00ULL) |
511  (orig >> 56));
512 }

519 static inline float ARSAL_Endianness_bswapf (float orig)
520 {
521  uint32_t res = ARSAL_Endianness_bswapl (*(uint32_t*)&orig);
522  return *(float *)&res;
523 }
